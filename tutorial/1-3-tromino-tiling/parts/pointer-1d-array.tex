\begin{frame}[fragile]{}
  \begin{definition}[Name of an Array]
    The \purple{value} of a \blue{variable of type array} is the \red{address} of \green{element zero} of the array.

    \[
      a \triangleq \&a[0]
    \]
  \end{definition}

  \begin{lstlisting}[style = Cstyle]
    int a[5];
    a, $\quad$ &a[0] // what are they?
    
    int *pa = a; 
    int *pa = &a[0];

    &a // what is this?
  \end{lstlisting}
