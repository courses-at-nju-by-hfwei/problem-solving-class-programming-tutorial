\begin{frame}[fragile]{Pass by Value}
  \begin{lstlisting}[style = Cstyle]
    swap(a, b);

    void swap(int a, int b) {
      int temp = a;
      a = b;
      b = tmp;
    }
  \end{lstlisting}

  \vspace{0.40cm}
  \pause
  \begin{lstlisting}[style = Cstyle]
    swap(&a, &b);

    void swap(int *a, int *b) {
      int temp = *a;
      *a = *b;
      *b = tmp;
    }
  \end{lstlisting}

  \vspace{0.30cm}
  \pause
  \stylebox{Place the * close to the variable name not pointer type.}
\end{frame}
