\begin{frame}[fragile]{\texttt{if; switch, case}}
  \begin{columns}
    \column{0.45\textwidth}
      \begin{lstlisting}[style = Cstyle]
  if (condition A) {
    ...
  } else if (condition B) {
    ...
  } else { // $\lnot A \lor \lnot B$
    ...
  }
      \end{lstlisting}
    \column{0.45\textwidth}
      \begin{lstlisting}[style = Cstyle]
  switch () {
    case c1:
      ...
      break;
    case c2:
      ...
      break;
    default:
  }
      \end{lstlisting}
  \end{columns}
\end{frame}
